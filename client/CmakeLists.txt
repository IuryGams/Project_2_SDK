add_executable(
    client

    src/client.cpp
)

    client

    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}/../include
        ${CMAKE_CURRENT_BINARY_DIR}/../proto
)

    client

    PUBLIC
        PROJECT_INTEGRATION_lib
        gRPC::grpc++
        protobuf::libprotobuf
)
